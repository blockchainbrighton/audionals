    <script>
        // Your JavaScript code to use ToneJS goes here

        document.addEventListener('tonejsLoaded', () => {
            // Update Status
            // document.querySelector('#status').textContent = `Status: Tone.js ${Tone.version} loaded. Play notes using the keyboard below.`;

            // Initialize Synth
            const synth = new Tone.Synth({
                oscillator: { type: 'sine' },
                envelope: {
                    attack: parseFloat(document.getElementById('attack').value),
                    decay: parseFloat(document.getElementById('decay').value),
                    sustain: parseFloat(document.getElementById('sustain').value),
                    release: parseFloat(document.getElementById('release').value)
                }
            }).toDestination();

            // Initialize Volume
            const volume = new Tone.Volume(parseFloat(document.getElementById('volume').value)).toDestination();
            synth.connect(volume);

            // Initialize Analyser for Visualization
            const analyser = new Tone.Analyser('waveform', 1024);
            synth.connect(analyser);

            // Update Synth Parameters Based on Controls
            document.getElementById('oscillatorType').addEventListener('change', (e) => {
                synth.oscillator.type = e.target.value;
            });

            document.getElementById('attack').addEventListener('input', (e) => {
                synth.envelope.attack = parseFloat(e.target.value);
            });

            document.getElementById('decay').addEventListener('input', (e) => {
                synth.envelope.decay = parseFloat(e.target.value);
            });

            document.getElementById('sustain').addEventListener('input', (e) => {
                synth.envelope.sustain = parseFloat(e.target.value);
            });

            document.getElementById('release').addEventListener('input', (e) => {
                synth.envelope.release = parseFloat(e.target.value);
            });

            document.getElementById('volume').addEventListener('input', (e) => {
                volume.volume.value = parseFloat(e.target.value);
            });

            // Keyboard Interaction
            const keys = document.querySelectorAll('.key');
            keys.forEach(key => {
                key.addEventListener('mousedown', () => {
                    const note = key.getAttribute('data-note');
                    synth.triggerAttack(note);
                    key.classList.add('active');
                });

                key.addEventListener('mouseup', () => {
                    const note = key.getAttribute('data-note');
                    synth.triggerRelease(note);
                    key.classList.remove('active');
                });

                key.addEventListener('mouseleave', () => {
                    const note = key.getAttribute('data-note');
                    synth.triggerRelease(note);
                    key.classList.remove('active');
                });
            });

            // Visualization
            const canvas = document.getElementById('waveform');
            const canvasCtx = canvas.getContext('2d');

            function draw() {
                requestAnimationFrame(draw);
                const buffer = analyser.getValue();
                canvasCtx.fillStyle = '#222';
                canvasCtx.fillRect(0, 0, canvas.width, canvas.height);
                canvasCtx.lineWidth = 2;
                canvasCtx.strokeStyle = '#28a745';
                canvasCtx.beginPath();

                const sliceWidth = canvas.width * 1.0 / buffer.length;
                let x = 0;

                for(let i = 0; i < buffer.length; i++) {
                    const v = buffer[i] / 255.0;
                    const y = v * canvas.height;

                    if(i === 0) {
                        canvasCtx.moveTo(x, y);
                    } else {
                        canvasCtx.lineTo(x, y);
                    }

                    x += sliceWidth;
                }

                canvasCtx.lineTo(canvas.width, canvas.height / 2);
                canvasCtx.stroke();
            }

            draw();
        });

        document.addEventListener('tonejsLoadError', (event) => {
            alert('Failed to load Tone.js. Please try again later.');
            console.error('Tone.js Load Error:', event.detail);
        });

        // Disable Play Button Initially
        // document.getElementById('playButton').disabled = true;
    </script>
