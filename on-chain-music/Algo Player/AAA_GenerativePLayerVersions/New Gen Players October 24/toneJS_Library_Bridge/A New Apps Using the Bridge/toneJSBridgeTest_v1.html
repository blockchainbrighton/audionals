            document.getElementById('status').textContent = `Status: Tone.js ${Tone.version} loaded. Click the button to play a note.`;

            // Initialize Synth
            const synth = new Tone.Synth().toDestination();

            // Enable Play Button
            const playButton = document.getElementById('playButton');
            playButton.disabled = false;
            playButton.addEventListener('click', () => {
                synth.triggerAttackRelease('C4', '8n');
            });
        });

        document.addEventListener('tonejsLoadError', (event) => {
            document.getElementById('status').textContent = `Status: Failed to load Tone.js`;
            console.error('Tone.js Load Error:', event.detail);
        });

        // Disable Play Button Initially
        document.getElementById('playButton').disabled = true;
